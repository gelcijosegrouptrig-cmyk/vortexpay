💡 Exemplo de Integração Fluxo completo de pagamento
JavaScript — Fluxo completo
const BASE = 'https://web-production-9f54e.up.railway.app';

// 1. Gerar cobrança Pix
const pix = await fetch(`${BASE}/api/pix`, {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ valor: 50.00 })
}).then(r => r.json());

console.log('Código Pix:', pix.pix_code);
console.log('TX ID:', pix.tx_id);

// 2. Verificar pagamento (polling)
const verificar = async (tx_id) => {
  const status = await fetch(`${BASE}/api/status/${tx_id}`).then(r => r.json());
  if (status.pago) {
    console.log('✅ Pago!');
  } else {
    setTimeout(() => verificar(tx_id), 15000); // retry em 15s
  }
};
verificar(pix.tx_id);
